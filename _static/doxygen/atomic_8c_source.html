180  real* n, real* n_0, int* enable_atomic) {

182 
183  /* Define a helper variable for storing rate coefficients, and
184  initialize the reaction rate variables */
185  real sigmav;
186  *rate_eff_rec = 0;
187  *rate_eff_ion = 0;
188 
189  /* Based on particle charge state, reaction rates are calculated for
190  possible reactions using the asigma helper module. Typically, asigma
191  does not return a reaction rate. Instead, it might for example return
192  a rate coefficient. A rate coefficient must be multiplied by the
193  density of the reaction counterpart to obtain the reaction rate.
194  NOTE: Currently, only two charge states and a limited range of
195  reaction types are implemented. */
196  if(q == 1) {
197  /* Test particle has charge state +1. Evaluate the rate coefficient
198  and multiply it by the reaction counterpart density to find the
199  reaction rate for a recombining (charge-decreasing) reaction. */
200  /* Charge exchange (CX)
201  NOTE: Ion density is not used in asigma_eval_sigmav() for
202  reac_type_sigmav_CX, so just pass density of main bulk ion species
203  as the dummy parameter. */
204  for(int i_spec = 0; i_spec < N_ntl_spec; i_spec++) {
205  err = asigma_eval_sigmav(&sigmav,
206  z_1, a_1, m_1,
207  z_2[i_spec], a_2[i_spec],
208  reac_type_sigmav_CX,
209  asigmadata,
210  E,
211  T[0], T_0[i_spec], n[1],
212  enable_atomic);
213  *rate_eff_rec += sigmav*n_0[i_spec];
214  }
215  } else if(q == 0) {
216  /* Test particle is neutral. Evaluate the beam-stopping (BMS)
217  coefficient. Because of how BMS coefficients are defined, to
218  determine the reaction rate, the BMS coefficient returned by
219  asigma_eval_sigmav() is multiplied by the electron density
220  corresponding to the ionic reaction counterpart density. The
221  species index maximum excludes electrons.
222  NOTE: Fully ionized ions are assumed when multiplying the effective
223  rate coefficient by the electron density corresponding to the ionic
224  reaction counterpart density.
225  NOTE: Neutral temperature is not used in asigma_eval_sigmav() for
226  reac_type_BMS_sigmav, so just pass temperature of main bulk neutral
227  species as the dummy parameter. */
228  for(int i_spec = 0; i_spec < (N_pls_spec-1); i_spec++) {
229  err = asigma_eval_sigmav(&sigmav,
230  z_1, a_1, m_1,
231  z_2[i_spec], a_2[i_spec],
232  reac_type_BMS_sigmav,
233  asigmadata,
234  E,
235  T[0], T_0[0], n[i_spec+1],
236  enable_atomic);
237  *rate_eff_ion += sigmav*z_2[i_spec]*n[i_spec+1];
238  }
239  } else {

265 a5err atomic_react(
266  int* q, real dt, real rate_eff_ion, real rate_eff_rec, int z_1, real rnd) {
267  a5err err = 0;
268 
269  /* Calculate the reaction probabilities for ionizing (charge-increasing)
270  and recombining (charge-decreasing) atomic reactions. But first,
271  a fail-safe against zero rate values. */
272  real prob_eff_ion;
273  real prob_eff_rec;
274  if(rate_eff_ion == 0.0) {
275  prob_eff_ion = 0.0;
276  } else {
277  prob_eff_ion = rate_eff_ion/(rate_eff_ion+rate_eff_rec)
278  *(1.0-exp(-(rate_eff_ion+rate_eff_rec)*dt));
279  }
280  if(rate_eff_rec == 0.0) {
281  prob_eff_rec = 0.0;
282  } else {
283  prob_eff_rec = rate_eff_rec/(rate_eff_ion+rate_eff_rec)
284  *(1.0-exp(-(rate_eff_ion+rate_eff_rec)*dt));
285  }
286 
287  /* Check if reaction occurs, and update charge state accordingly */
288  if(*q < z_1 && rnd < prob_eff_ion) {
289  *q += 1;
290  } else if(*q > 0 && rnd < (prob_eff_ion+prob_eff_rec)) {
291  *q -= 1;
292  }
293 
294  return err;
295 }
