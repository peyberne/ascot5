351  psi1 = Bdata->BSTS.psi1;
352  break;
353 
354  case B_field_type_TC:
355  psi0 = Bdata->BTC.psival;
356  psi1 = 2.0;
357  break;
358 
359  default:
360  /* Unregonized input. Produce error. */
361  err = error_raise( ERR_UNKNOWN_INPUT, __LINE__, EF_B_FIELD );
362  break;
363  }
364

366  if( (psi - psi0) / delta < 0 ) {
367  err = error_raise( ERR_INPUT_UNPHYSICAL, __LINE__, EF_B_FIELD );
368  } else {
369  rho[0] = sqrt( (psi - psi0) / delta );
370  rho[1] = 1.0 / (2*delta*rho[0]);
371  }
372

374  /* In case of error, return some reasonable value to avoid further
375  complications */
376  rho[0] = 1;
377  rho[1] = 0;
