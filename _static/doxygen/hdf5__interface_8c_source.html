729 void hdf5_generate_qid(char* qid) {
730 
731  /* Seed random number generator with current time */
732  struct timespec ts;
733 #ifdef __MACH__
734  clock_gettime(CLOCK_MONOTONIC_RAW, &ts);
735 #else
736  timespec_get(&ts, TIME_UTC);
737 #endif
738  srand48( ts.tv_nsec );
739 
740  /* Generate a 32 bit random integer by generating signed 32 bit random
741  * integers with mrand48() and choosing the first one that is positive */
742  long int qint = -1;
743  while(qint < 0) {
744  qint = mrand48();
745  }
746 
747  /* Convert the random number to a string format */
748  sprintf(qid, "%010lu", (long unsigned int)qint);
