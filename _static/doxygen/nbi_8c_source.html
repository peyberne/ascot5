143  /* One marker contributes to a distribution as weight*dt. If beams are on
144  * for a set time, use the time interval as dt. If beams are on indefinitely
145  * (t1=t0) then use dt=1.0 s since then we don't need to normalize the
146  * distribution afterwards */

216  ptemp.charge = charge;

241  * generate markers until the particle struct is filled (skip this loop
242  * if markers are not requested). We have do it like this because we can

307 void nbi_inject(real* xyz, real* vxyz, real* efrac, nbi_injector* inj,
308  random_data* rng) {
309  /* Pick a random beamlet and initialize marker there */
310  int i_beamlet = floor(random_uniform(rng) * inj->n_beamlet);
311  xyz[0] = inj->beamlet_x[i_beamlet];
312  xyz[1] = inj->beamlet_y[i_beamlet];
313  xyz[2] = inj->beamlet_z[i_beamlet];
314 
315  /* Pick marker energy based on the energy fractions */
316  real energy;
317  real r = random_uniform(rng);
318  if(r < inj->efrac[0]) {
319  energy = inj->energy;
320  *efrac = 1.0;
321  } else if(r < inj->efrac[0] + inj->efrac[1]) {
322  energy = inj->energy / 2;
323  *efrac = 1.0/2;
324  } else {
325  energy = inj->energy / 3;
326  *efrac = 1.0/3;
327  }
328 
329  /* Calculate vertical and horizontal normals for beam divergence */
330  real dir[3], normalv[3], normalh[3], tmp[3];
331  dir[0] = inj->beamlet_dx[i_beamlet];
332  dir[1] = inj->beamlet_dy[i_beamlet];
333  dir[2] = inj->beamlet_dz[i_beamlet];
334 
335  real phi = atan2(dir[1], dir[0]);
336  real theta = acos(dir[2]);
337 
338  normalv[0] = sin(theta+CONST_PI/2) * cos(phi);
339  normalv[1] = sin(theta+CONST_PI/2) * sin(phi);
340  normalv[2] = cos(theta+CONST_PI/2);
341 
342  math_cross(dir, normalv, tmp);
343  math_unit(tmp, normalh);
344 
345  /* Assuming isotropic divergence using horizontal value,
346  ignoring halo divergence for now */
347  r = random_uniform(rng);
348  real div = inj->div_h * sqrt(log(1/(1-r)));
349 
350  real angle = random_uniform(rng) * 2 * CONST_PI;
351  tmp[0] = dir[0] + div * ( cos(angle)*normalh[0] + sin(angle)*normalv[0] );
352  tmp[1] = dir[1] + div * ( cos(angle)*normalh[1] + sin(angle)*normalv[1] );
353  tmp[2] = dir[2] + div * ( cos(angle)*normalh[2] + sin(angle)*normalv[2] );
354 
355  math_unit(tmp, dir);
356 
357  real gamma = physlib_gamma_Ekin(inj->mass, energy);
358  real absv = sqrt( 1.0 - 1.0 / (gamma * gamma) ) * CONST_C;
359  vxyz[0] = absv * dir[0];
360  vxyz[1] = absv * dir[1];
361  vxyz[2] = absv * dir[2];
362 }

382 void nbi_ionize(real* xyz, real* vxyz, real time, int* shinethrough, int anum,
383  int znum, real mass, B_field_data* Bdata, plasma_data* plsdata,
384  wall_data* walldata, random_data* rng) {
385  a5err err;
386 
387  real vnorm = math_norm(vxyz);
388  real gamma = physlib_gamma_vnorm(vnorm);
389  real ekin = physlib_Ekin_gamma(mass, gamma);
390  real vhat[3];
391  vhat[0] = vxyz[0] / vnorm;
392  vhat[1] = vxyz[1] / vnorm;
393  vhat[2] = vxyz[2] / vnorm;
394 
395  int n_species = plasma_get_n_species(plsdata);
396  const int* pls_anum = plasma_get_species_anum(plsdata);
397  const int* pls_znum = plasma_get_species_znum(plsdata);
398  real pls_temp[MAX_SPECIES], pls_dens[MAX_SPECIES];
399 
400  real remaining = 1;
401  real threshold = random_uniform(rng);
402  real ds = 1e-3;
403 
404  real s = 0.0;
405  int entered_plasma = 0;
406  int exited_plasma = 0;
407 
408  while(remaining > threshold && s < NBI_MAX_DISTANCE) {
409  err = 0;
410  real rate = 0.0;
411 
412  real rpz[3];
413  math_xyz2rpz(xyz, rpz);
414 
415  real psi, rho[2];
416  err = B_field_eval_psi(&psi, rpz[0], rpz[1], rpz[2], 0.0, Bdata);
417 
418  if(!err) {
419  err = B_field_eval_rho(rho, psi, Bdata);
420 
421  /* Check for wall collisions after passing through separatrix
422  * twice */
423  if(!entered_plasma && rho[0] < 1.0) {
424  entered_plasma = 1;
425  }
426  if(entered_plasma && !exited_plasma && rho[0] >= 1.0) {
427  exited_plasma = 1;
428  }
429 
430  err = plasma_eval_densandtemp(pls_dens, pls_temp, rho[0], rpz[0],
431  rpz[1], rpz[2], time, plsdata);
432 
433  if(!err) {
434  real sigmav;
435  if( suzuki_sigmav(
436  &sigmav, ekin / anum, pls_dens[0], pls_temp[0],
437  n_species-1, &(pls_dens[1]), pls_anum, pls_znum) ) {
438  err = 1;
439  }
440  rate = pls_dens[0] * sigmav;
441  }
442  else {
443  rate = 0.0; /* Outside the plasma */
444  }
445  }
446  else {
447  rate = 0.0; /* Outside the magnetic field */
448  }
449 
450  s += ds;
451  xyz[0] += ds * vhat[0];
452  xyz[1] += ds * vhat[1];
453  xyz[2] += ds * vhat[2];
454  remaining *= exp(-rate * ds);
455 
456  if(exited_plasma) {
457  real rpz2[3]; /* New position, old position already in rpz */
458  math_xyz2rpz(xyz, rpz2);
459  real w_coll;
460  int tile = wall_hit_wall(rpz[0], rpz[1], rpz[2], rpz2[0], rpz2[1],
461  rpz2[2], walldata, &w_coll);
462 
463  if(tile > 0) {
464  /* Hit wall */
465  *shinethrough = tile;
466  return;
467  }
468  }
469  }
470 
471  /* Check if we've missed the plasma entirely */
472  if(s < NBI_MAX_DISTANCE) {
473  *shinethrough = 0;
474  }
475  else {
476  *shinethrough = -1;
477  }
478 }
