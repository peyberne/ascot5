425  if(!reac_found) {
426  /* Reaction not found. Raise error. */
427  err = error_raise( ERR_INPUT_EVALUATION, __LINE__,

437  int interperr = 0;

439  &asgm_loc_data->sigma[i_reac],
440  E_coll_per_amu);
441  if(interperr) {
442  /* Energy is outside spline domain */
443  if(*enable_atomic == 2) {
444  /* Set cross-section to 0 to avoid further problems */
445  *sigma = 0.0;

447  /* Raise an error */

520  reac_found = 1;
521  break;
522  }
523  }
524 
525  /* The rate coefficient is evaluated if reaction data was found,
526  is available, and its interpolation works. */
527  if(!reac_found) {
528  /* Reaction not found. Raise error. */
529  err = error_raise( ERR_INPUT_EVALUATION, __LINE__,
