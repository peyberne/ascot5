48  /* Calculate bicubic spline surface coefficients. For each grid cell
49  (i_x, i_y), there are 16 coefficients, one for each variable product
50  dx^p_x*dy^p_y in the evaluation formula, where p_x, p_y = 0, 1, 2, 3. */
51 
52  /* Cubic spline along x for each y, using f values to get a total of four
53  coefficients */
54  for(int i_y=0; i_y<n_y; i_y++) {
55  for(int i_x=0; i_x<n_x; i_x++) {
56  f_x[i_x] = f[i_y*n_x+i_x];
57  }
58  splineexpl(f_x, n_x, bc_x, c_x);
59  for(int i_x=0; i_x<n_x-1; i_x++) {
60  for(int i_c=0; i_c<4; i_c++) {
61  c[i_y*n_x*16+i_x*16+i_c] = c_x[i_x*4+i_c];
62  }
63  }
64  }
65 
66  /* Four cubic splines along y for each x, using the above calculated four
67  coefficient values to get a total of 16 coefficients */
68  for(int i_x=0; i_x<n_x-1; i_x++) {
69  for(int i_s=0; i_s<4; i_s++) {
70  for(int i_y=0; i_y<n_y; i_y++) {
71  f_y[i_y] = c[i_y*n_x*16+i_x*16+i_s];
72  }
73  splineexpl(f_y, n_y, bc_y, c_y);
74  for(int i_y=0; i_y<n_y-1; i_y++) {
75  i_ct = 0;
76  for(int i_c=i_s; i_c<16; i_c=i_c+4) {
77  c[i_y*n_x*16+i_x*16+i_c] = c_y[i_y*4+i_ct];
78  i_ct++;
79  }
80  }
81  }
82  }
83
