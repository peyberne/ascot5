326 void afsi_compute_product_momenta(
327  int i, real m1, real m2, real mprod1, real mprod2, real Q,
328  real* v1x, real* v1y, real* v1z, real* v2x, real* v2y, real* v2z,
329  real* ppara1, real* pperp1, real* ppara2, real* pperp2) {
330 
331  // Velocity of the system's center of mass
332  real v_cm[3];
333  v_cm[0] = ( m1 * v1x[i] + m2 * v2x[i] ) / ( m1 + m2 );
334  v_cm[1] = ( m1 * v1y[i] + m2 * v2y[i] ) / ( m1 + m2 );
335  v_cm[2] = ( m1 * v1z[i] + m2 * v2z[i] ) / ( m1 + m2 );
336 
337  // Total kinetic energy after the reaction in CM frame
338  real ekin = Q
339  + 0.5 * m1 * ( (v1x[i] - v_cm[0])*(v1x[i] - v_cm[0])
340  + (v1y[i] - v_cm[1])*(v1y[i] - v_cm[1])
341  + (v1z[i] - v_cm[2])*(v1z[i] - v_cm[2]) )
342  + 0.5 * m2 * ( (v2x[i] - v_cm[0])*(v2x[i] - v_cm[0])
343  + (v2y[i] - v_cm[1])*(v2y[i] - v_cm[1])
344  + (v2z[i] - v_cm[2])*(v2z[i] - v_cm[2]) );
345 
346  // Speed and velocity of product 2 in CM frame
347  real rn1 = random_uniform(rdata);
348  real rn2 = random_uniform(rdata);
349  real phi = CONST_2PI * rn1;
350  real theta = acos( 2 * ( rn2 - 0.5 ) );
351  real vnorm = sqrt( 2.0 * ekin / ( mprod2 * ( 1.0 + mprod2 / mprod1 ) ) );
352 
353  real v2_cm[3];
354  v2_cm[0] = vnorm * sin(theta) * cos(phi);
355  v2_cm[1] = vnorm * sin(theta) * sin(phi);
356  v2_cm[2] = vnorm * cos(theta);
357 
358  // Products' velocities in lab frame
359  real vprod1[3], vprod2[3];
360  vprod1[0] = -(mprod2/mprod1) * v2_cm[0] + v_cm[0];
361  vprod1[1] = -(mprod2/mprod1) * v2_cm[1] + v_cm[1];
362  vprod1[2] = -(mprod2/mprod1) * v2_cm[2] + v_cm[2];
363  vprod2[0] = v2_cm[0] + v_cm[0];
364  vprod2[1] = v2_cm[1] + v_cm[1];
365  vprod2[2] = v2_cm[2] + v_cm[2];
366 
367  // ppara and pperp
368  ppara1[i] = vprod1[2] * mprod1;
369  pperp1[i] = sqrt( vprod1[0]*vprod1[0] + vprod1[1]*vprod1[1] ) * mprod1;
370  ppara2[i] = vprod2[2] * mprod2;
371  pperp2[i] = sqrt( vprod2[0]*vprod2[0] + vprod2[1]*vprod2[1] ) * mprod2;
372 }

385 void afsi_sample_5D(dist_5D_data* dist, int n, int iR, int iphi, int iz,
386  real* ppara, real* pperp) {

388 
389  for(int ippara = 0; ippara < dist->n_ppara; ippara++) {
390  for(int ipperp = 0; ipperp < dist->n_pperp; ipperp++) {
391  if(ippara == 0 && ipperp == 0) {
392  cumdist[0] = dist->histogram[dist_5D_index(iR, iphi, iz,
393  0, 0, 0, 0, dist->n_phi, dist->n_z, dist->n_ppara,
394  dist->n_pperp, 1, 1)];
395  } else {
396  cumdist[ippara*dist->n_pperp+ipperp] =
397  cumdist[ippara*dist->n_pperp+ipperp-1]
398  + dist->histogram[dist_5D_index(iR, iphi, iz,
399  ippara, ipperp, 0, 0, dist->n_phi, dist->n_z,
400  dist->n_ppara, dist->n_pperp, 1, 1)];
401  }
402  }
403  }
404  for(int ippara = 0; ippara < dist->n_ppara; ippara++) {
405  for(int ipperp = 0; ipperp < dist->n_pperp; ipperp++) {
406  cumdist[ippara*dist->n_pperp+ipperp] /=
407  cumdist[dist->n_ppara*dist->n_pperp-1];
408  }
409  }

438 void afsi_sample_thermal(afsi_thermal_data* data, real mass, int n, int iR,
439  int iphi, int iz, real* ppara, real* pperp) {
440  int ind = iR * (data->n_phi * data->n_z) + iphi * data->n_z + iz;
441  real temp = data->temperature[ind];
442 
443  for(int i = 0; i < n; i++) {
444  real w = 0;
445  real r1;
446  real r2;
447 
448  while(w <= 1.0) {
449  r1 = random_uniform(rdata);
450  r2 = random_uniform(rdata);
451  w = sqrt(r1*r1 + r2*r2);
452  }
453 
454  real r3 = random_uniform(rdata);
455  real r4 = random_uniform(rdata);
456 
457  real E = -temp * (r1*r1 * log(r3) / w + log(r4));
458  real absv = sqrt(2*E*CONST_E/mass);
459 
460  r1 = random_uniform(rdata);
461  r2 = random_uniform(rdata);
462 
463  real theta = CONST_2PI * r1;
464  real phi = acos(1 - 2*r2);
465  real vx = absv * sin(phi) * cos(theta);
466  real vy = absv * sin(phi) * sin(theta);
467  real vz = absv * cos(phi);
468 
469  ppara[i] = vz * mass;
470  pperp[i] = sqrt(vx*vx + vy*vy) * mass;
471  }
