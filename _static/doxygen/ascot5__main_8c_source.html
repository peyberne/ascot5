487  int n_host = nprts;
488 #endif

517  int_offload_array, *diag_offload_array);
518 
519  mic_end = omp_get_wtime();
520  }
521 #endif
522 
523  /* No target, marker simulation happens where the code execution began.
524  * Offloading is only emulated. */
525 #ifndef TARGET
526  #pragma omp section

528  host_start = omp_get_wtime();
529  simulate(0, n_host, pin+2*n_mic, sim, offload_data,

532  }
533 #endif
534  }
535 
536  /* Code execution returns to host. */
537  print_out0(VERBOSE_NORMAL, mpi_rank, "gpu %lf s, host %lf s\n",
538  mic_end-mic_start, host_end-host_start);
539 
540  /* Gather output data */
541  mpi_gather_particlestate(pin, pout, n_gathered, n_tot,
542  mpi_rank, mpi_size, mpi_root);
543  free(pin);
544 
545  mpi_gather_diag(&sim->diag_offload_data, *diag_offload_array, n_tot,
546  mpi_rank, mpi_size, mpi_root);
547 
548  return 0;
549 }

793 void print_marker_summary(particle_state* ps, int n_tot) {
794 
795  print_out(VERBOSE_MINIMAL, "\nSummary of results:\n");
796 
797  /* Temporary arrays that are needed to store unique end conditions and
798  * errors. We can have at most n_tot different values. */
799  int* temp = (int*)malloc(n_tot*sizeof(int));
800  int* unique = (int*)malloc(n_tot*sizeof(int));
801  int* count = (int*)malloc((n_tot+1)*sizeof(int));
802 
803  /* First we find out what the end conditions are */
804  for(int i=0; i<n_tot; i++) {
805  temp[i] = ps[i].endcond;
806  }
807  math_uniquecount(temp, unique, count, n_tot);
808  count[n_tot] = 0;/* This ensures the following while loops are terminated.*/
809 
810  /* Print the end conditions, if marker has multiple end conditions, separate
811  * those with "and". */
812  int i = 0;
813  while(count[i] > 0) {
814  /* Initialize */
815  int endconds[32];
816  for(int j=0; j<32;j++) {
817  endconds[j] = 0;
818  }
819  char endcondstr[256];
820  endcondstr[0] = '\0';
821 
822  /* Represent all end conditions with a single string and print it */
823  int j = 0;
824  endcond_parse(unique[i], endconds);
825  while(endconds[j]) {
826  if(j>0) {
827  strcat(endcondstr, " and ");
828  }
829  char temp[256];
830  endcond_parse2str(endconds[j], temp);
831  strcat(endcondstr, temp);
832  j++;
833  }
834  if(j == 0) {
835  sprintf(endcondstr, "Aborted");
836  }
837  print_out(VERBOSE_MINIMAL, "%9d markers had end condition %s\n",
838  count[i], endcondstr);
839  i++;
840  }
841 
842  /* Empty line between end conditions and errors */
843  print_out(VERBOSE_MINIMAL, "\n");
844 
845  /* Find all errors */
846  for(int i=0; i<n_tot; i++) {
847  temp[i] = (int)(ps[i].err);
848  }
849  math_uniquecount(temp, unique, count, n_tot);
850 
851  /* Go through each unique error and represent is a string */
852  i = 0;
853  while(count[i] > 0) {
854  if(unique[i] == 0) {
855  /* Value 0 indicates no error occurred so skip that */
856  i++;
857  continue;
858  }
859  char msg[256];
860  char line[256];
861  char file[256];
862  error_parse2str(unique[i], msg, line, file);
863  print_out(VERBOSE_MINIMAL,
864  "%9d markers were aborted with an error message:\n"
865  " %s\n"
866  " at line %s in %s\n",
867  count[i], msg, line, file);
868  i++;
869  }
870 
871  /* If count[0] equals to number of markers and their error field is zero,
872  * we have no markers that were aborted. */
873  if(count[0] == n_tot && unique[0] == 0) {
874  print_out(VERBOSE_MINIMAL,
875  " No markers were aborted.\n");
876  }
877 
878  /* Free temporary arrays */
879  free(temp);
880  free(unique);
881  free(count);
882 }
