292  B_2DS_data* Bdata) {

302 
303  /* Check that the values seem valid */

306  err = error_raise( ERR_INPUT_UNPHYSICAL, __LINE__, EF_B_2DS );

310  rho_drho[0] = sqrt((psi_dpsi[0] - Bdata->psi0) / delta);

312  rho_drho[2] = 0;
