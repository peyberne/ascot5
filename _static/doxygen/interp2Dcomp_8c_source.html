172  x = x + (x < str->x_min) * (str->x_max - str->x_min);
173  }

180  int i_x = (x - str->x_min) / str->x_grid - 1*(x==str->x_max);
181  /* Normalized x coordinate in current cell */
182  real dx = ( x - (str->x_min + i_x*str->x_grid) ) / str->x_grid;
183  /* Helper variables */
184  real dx3 = dx * (dx*dx - 1.0);
185  real dxi = 1.0 - dx;
186  real dxi3 = dxi * (dxi*dxi - 1.0);

188
