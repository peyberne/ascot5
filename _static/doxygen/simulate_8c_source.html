399 void sim_monitor(char* filename, volatile int* n, volatile int* finished) {
400  /* Open a file for writing simulation progress */
401  FILE *f = fopen(filename, "w");
402  if (f == NULL) {
403  print_out(VERBOSE_DEBUG,
404  "Warning. %s could not be opened for progress updates.\n",
405  filename);
406  return;
407  }
408 
409  real time_sim_started = A5_WTIME;
410  int stopflag = 1; /* Flag ensures progress is written one last time at 100% */
411  int n_temp, finished_temp; /* Use these to store volatile variables so that
412  their value does not change during one loop */
413  while(stopflag) {
414  n_temp = *n;
415  finished_temp = *finished;
416  real fracprog = ((real) finished_temp)/n_temp;
417  real timespent = (A5_WTIME)-time_sim_started;
418 
419  if(n_temp == finished_temp) {
420  stopflag = 0;
421  }
422 
423  if(fracprog == 0) {
424  fprintf(f, "No marker has finished simulation yet. "
425  "Time spent: %.2f h\n", timespent/3600);
426  }
427  else {
428  fprintf(f, "Progress: %d/%d, %.2f %%. Time spent: %.2f h, "
429  "estimated time to finish: %.2f h\n", finished_temp, n_temp,
430  100*fracprog, timespent/3600, (1/fracprog-1)*timespent/3600);
431  }
432  fflush(f);
433  sleep(A5_PRINTPROGRESSINTERVAL);
434  }
435 
436  fprintf(f, "Simulation finished.\n");
437  fclose(f);
438 }
