176  integer id, real rho, real r, real pitchsign,
177  real t_f, real t_i, real theta_f, real theta_i) {
178  /* Mask dummy markers */
179  if( id > 0 ) {
180 
181  /* Check whether marker position should be recorded: *
182  * - Time step was accepted t_f > t_i
183  * - Enough time has passed since last record t_f - ti > interval OR
184  no data exists yet.
185  * - Marker has crossed OMP during the current time step.
186  */
187  real record = 0.0;
188  if( t_f > t_i ) {
189  if( data->datapoints[index] == NULL ) {
190  record = diag_transcoef_check_omp_crossing(theta_f, theta_i);
191  }
192  else if( t_f - data->datapoints[index]->time > data->interval ) {
193  record = diag_transcoef_check_omp_crossing(theta_f, theta_i);
194  }
195  }
196 
197  /* Record */
198  if( record > 0) {
199  diag_transcoef_link* newlink =

201  newlink->time = t_f;
202  newlink->pitchsign = pitchsign;
203 
204  if(data->recordrho) {

209  }

234 void diag_transcoef_process_and_clean(diag_transcoef_data* data,
235  integer index, integer id) {
236 
237  /* Count number of positive and negative crossings */
238  int positive = 0, negative = 0;
239  diag_transcoef_link* link = data->datapoints[index];
240  while(link != NULL) {
241  if(link->pitchsign < 0) {
242  negative++;
243  }
244  else {
245  positive++;
246  }
247  link = link->prevlink;
248  }
249 
250  /* Which ever there are more are stored */
251  int datasize, sign;
252  if(positive >= negative) {
253  datasize = positive;
254  sign = 1;
255  }
256  else {
257  datasize = negative;
258  sign = -1;
259  }
260 
261  /* If there are enough datapoints, process them to K and D */
262  if(datasize > data->Navg) {
263  /* How many points we have after averaging data */
264  int navgpnt = ceil(datasize/data->Navg);
265  real* rho = malloc(navgpnt*sizeof(real));
266  real* time = malloc(navgpnt*sizeof(real));
267 
268  /* The datasize is not necessarily multiple of navg. We take the
269  "extra" points from last points and average them separately */
270  link = data->datapoints[index];
271  while(link->pitchsign * sign < 0) {
272  link = link->prevlink;
273  }
274  rho[navgpnt-1] = link->rho;
275  time[navgpnt-1] = link->time;
276 
277  int nextrapnts = datasize - navgpnt*data->Navg;
278  if(nextrapnts == 0) {
279  nextrapnts = data->Navg;
280  }
281  for(int k = 1; k < nextrapnts; k++) {
282  link = link->prevlink;
283  while(link->pitchsign * sign < 0) {
284  link = link->prevlink;
285  }
286  rho[navgpnt-1] += link->rho;
287  time[navgpnt-1] += link->time;
288  }
289  rho[navgpnt-1] /= nextrapnts;
290  time[navgpnt-1] /= nextrapnts;
291 
292  /* Take the full averages */
293  for(int j = navgpnt-2; j > -1; j--) {
294  rho[j] = 0;
295  time[j] = 0;
296  for(int k = 0; k < data->Navg; k++) {
297  link = link->prevlink;
298  while(link->pitchsign * sign < 0) {
299  link = link->prevlink;
300  }
301  rho[j] += link->rho;
302  time[j] += link->time;
303  }
304  rho[j] /= data->Navg;
305  time[j] /= data->Navg;
306  }
307 
308  /* Evaluate coefficients */
309  real K = 0;
310  for(int j = 0; j < navgpnt-1; j++) {
311  K += ( rho[j+1] - rho[j] ) / ( time[j+1] - time[j] );
312  }
313  K = K / navgpnt;
314 
315  real D = 0;
316  for(int j = 0; j < navgpnt-1; j++) {
317  real a = rho[j+1] - rho[j] - K * ( time[j+1] - time[j] );
318  D += 0.5*a*a / ( time[j+1] - time[j] );
319  }
320  D = D / navgpnt;
321 
322  /* Store and free resources */
323  free(rho);
324  free(time);
325  data->id[index] = (real)id;
326  data->Kcoef[index] = K;
327  data->Dcoef[index] = D;
328  }
329 
330  /* Clear temporary storage */
331  diag_transcoef_link* temp;
332  link = data->datapoints[index];
333  while(link != NULL) {
334  temp = link->prevlink;
335  free(link);
336  link = temp;
337  }
338  data->datapoints[index] = NULL;
339 
340 }
